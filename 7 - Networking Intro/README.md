            if location.startswith('https://'):
                print("‚úÖ –†–µ–¥–∏—Ä–µ–∫—Ç –Ω–∞ –±–µ–∑–ø–µ—á–Ω–∏–π HTTPS")

        elif response.status_code == 403:
            print("‚õî –î–æ—Å—Ç—É–ø –∑–∞–±–æ—Ä–æ–Ω–µ–Ω–æ")
            
        elif response.status_code == 404:
            print("‚ùå –°—Ç–æ—Ä—ñ–Ω–∫–∞ –Ω–µ –∑–Ω–∞–π–¥–µ–Ω–∞")

            print(f"üí• –ü–æ–º–∏–ª–∫–∞ —Å–µ—Ä–≤–µ—Ä–∞: {response.status_code}")
            
    except requests.exceptions.SSLError:

### 2.6. –¢–∞–π–º–∞—É—Ç–∏ —Ç–∞ –æ–±—Ä–æ–±–∫–∞ –ø–æ–º–∏–ª–æ–∫

```python
import requests
from requests.exceptions import (
    Timeout, 
    ConnectionError, 
    HTTPError,
    RequestException
)

def safe_request(url, timeout=5, max_retries=3):
    """
    –ë–µ–∑–ø–µ—á–Ω–µ –≤–∏–∫–æ–Ω–∞–Ω–Ω—è HTTP –∑–∞–ø–∏—Ç—É –∑ –ø–æ–≤—Ç–æ—Ä–Ω–∏–º–∏ —Å–ø—Ä–æ–±–∞–º–∏
    """
    for attempt in range(max_retries):
        try:
            response = requests.get(url, timeout=timeout)
            response.raise_for_status()  # –í–∏–∫–∏–Ω—É—Ç–∏ –≤–∏–∫–ª—é—á–µ–Ω–Ω—è –¥–ª—è 4xx/5xx
            return response
            
        except Timeout:
            print(f"‚è±Ô∏è –°–ø—Ä–æ–±–∞ {attempt + 1}: –¢–∞–π–º-–∞—É—Ç")
            if attempt == max_retries - 1:
                raise
                
        except ConnectionError:
            print(f"‚ùå –°–ø—Ä–æ–±–∞ {attempt + 1}: –ü–æ–º–∏–ª–∫–∞ –∑'—î–¥–Ω–∞–Ω–Ω—è")
            if attempt == max_retries - 1:
                raise
                
        except HTTPError as e:
            print(f"‚ùå HTTP –ø–æ–º–∏–ª–∫–∞: {e.response.status_code}")
            raise
            
        except RequestException as e:
            print(f"‚ùå –ó–∞–≥–∞–ª—å–Ω–∞ –ø–æ–º–∏–ª–∫–∞: {e}")
            raise

# –í–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–Ω—è
try:
    response = safe_request('https://api.example.com/data', timeout=10)
    print(f"‚úÖ –£—Å–ø—ñ—Ö: {response.status_code}")
except Exception as e:
    print(f"üí• –ù–µ –≤–¥–∞–ª–æ—Å—è –≤–∏–∫–æ–Ω–∞—Ç–∏ –∑–∞–ø–∏—Ç: {e}")
```

        """
        –°–∫–∞–Ω—É–≤–∞–Ω–Ω—è URL –Ω–∞ –Ω–∞—è–≤–Ω—ñ—Å—Ç—å –∑–∞–≥–æ–ª–æ–≤–∫—ñ–≤ –±–µ–∑–ø–µ–∫–∏
        """

            'url': url,
            'timestamp': datetime.now().isoformat(),
            'status': 'unknown',
            'server': 'unknown',
            'headers_found': [],
            'headers_missing': [],
            'score': 0,
            'max_score': 100
        }
        
        try:
            response = requests.get(
                url,
                timeout=self.timeout,
                allow_redirects=True
            )
            
            result['status'] = response.status_code
            result['server'] = response.headers.get('Server', 'unknown')
            
            # –ü–µ—Ä–µ–≤—ñ—Ä–∫–∞ –∫–æ–∂–Ω–æ–≥–æ –∑–∞–≥–æ–ª–æ–≤–∫–∞
            for header, info in self.SECURITY_HEADERS.items():
                if header in response.headers:
                    result['headers_found'].append({
                        'header': header,
                        'value': response.headers[header],
                        'severity': info['severity'],
                        'description': info['description']
                    })
                else:
                    result['headers_missing'].append({
                        'header': header,
                        'severity': info['severity'],
                        'description': info['description']
                    })
            
            # –†–æ–∑—Ä–∞—Ö—É–Ω–æ–∫ –±–∞–ª—É
            total_headers = len(self.SECURITY_HEADERS)
            found_headers = len(result['headers_found'])
            result['score'] = int((found_headers / total_headers) * 100)
            
        except requests.exceptions.RequestException as e:
            result['error'] = str(e)
        
        return result
    
    def print_report(self, result: Dict):

def json_to_xml(json_data, root_name='root'):
    """
    –ö–æ–Ω–≤–µ—Ä—Ç–∞—Ü—ñ—è JSON –≤ XML
    """
    def build_xml(data, parent):
        if isinstance(data, dict):
            for key, value in data.items():
                child = ET.SubElement(parent, key)
                build_xml(value, child)
        elif isinstance(data, list):
            for item in data:
                child = ET.SubElement(parent, 'item')
                build_xml(item, child)
        else:

    build_xml(json_data, root)
    return ET.tostring(root, encoding='unicode')

            if 'cvssMetricV31' in metrics:
                cvss_data = metrics['cvssMetricV31'][0]['cvssData']
                cvss_score = cvss_data.get('baseScore', 0.0)
            
            # –î–∞—Ç–∞ –ø—É–±–ª—ñ–∫–∞—Ü—ñ—ó
            published = cve.get('published', 'Unknown')
            
            vulnerabilities.append({
                'cve_id': cve_id,
                'description': description[:200] + '...' if len(description) > 200 else description,
                'cvss_score': cvss_score,
                'severity': self.get_severity(cvss_score),
                'published': published
            })
        
        return vulnerabilities
    
    @staticmethod
    def get_severity(cvss_score: float) -> str:
        """
        –í–∏–∑–Ω–∞—á–µ–Ω–Ω—è severity –∑–∞ CVSS score
        """
        if cvss_score >= 9.0:
            return 'CRITICAL'
        elif cvss_score >= 7.0:
            return 'HIGH'
        elif cvss_score >= 4.0:
            return 'MEDIUM'
        elif cvss_score > 0:
            return 'LOW'
        else:
            return 'NONE'
    
    def print_report(self, vulnerabilities: List[Dict]):
        """
