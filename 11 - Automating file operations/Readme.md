### **–°–ª–∞–π–¥ 4. Lifehack: –°–æ—Ä—Ç—É–≤–∞–Ω–Ω—è –∑–∞ —á–∞—Å–æ–º**

```python
from pathlib import Path
import datetime

def get_timestamp_from_filename(filename):
    """–í–∏—Ç—è–≥—Ç–∏ timestamp –∑ –Ω–∞–∑–≤–∏ —Ñ–∞–π–ª—É"""
    # –ü—Ä–∏–ø—É—Å—Ç–∏–º–æ —Ñ–æ—Ä–º–∞—Ç: backup_2024-11-09_14-30-45.zip
    parts = filename.stem.split('_')
    if len(parts) >= 3:
        date_part = parts[-2]  # 2024-11-09
        time_part = parts[-1]  # 14-30-45
        timestamp_str = f"{date_part} {time_part.replace('-', ':')}"
        return datetime.datetime.strptime(timestamp_str, "%Y-%m-%d %H:%M:%S")
    return None

# –ó–Ω–∞–π—Ç–∏ –≤—Å—ñ backup —Ñ–∞–π–ª–∏
backups = list(Path("backups").glob("backup_*.zip"))

# –°–æ—Ä—Ç—É–≤–∞–Ω–Ω—è –∑–∞ —á–∞—Å–æ–º (–≤—ñ–¥ –Ω–∞–π—Å—Ç–∞—Ä—ñ—à–æ–≥–æ –¥–æ –Ω–∞–π–Ω–æ–≤—ñ—à–æ–≥–æ)
sorted_backups = sorted(backups, key=lambda x: get_timestamp_from_filename(x.name))

print("–†–µ–∑–µ—Ä–≤–Ω—ñ –∫–æ–ø—ñ—ó (—Ö—Ä–æ–Ω–æ–ª–æ–≥—ñ—á–Ω–æ):")
for backup in sorted_backups:
    ts = get_timestamp_from_filename(backup.name)
    print(f"  {backup.name} - {ts}")
```

    –ö–æ–ø—ñ—é–≤–∞—Ç–∏ —Ç—ñ–ª—å–∫–∏ —Ñ–∞–π–ª–∏, –∑–º—ñ–Ω–µ–Ω—ñ –∑–∞ –æ—Å—Ç–∞–Ω–Ω—ñ N –¥–Ω—ñ–≤
    """
    source = Path(source_dir)
    backup = Path(backup_dir)
    backup.mkdir(exist_ok=True)

    cutoff_time = time.time() - (days * 24 * 60 * 60)
    copied_count = 0

    for file_path in source.rglob("*"):
        if file_path.is_file():
            # –ü–µ—Ä–µ–≤—ñ—Ä–∏—Ç–∏ —á–∞—Å –º–æ–¥–∏—Ñ—ñ–∫–∞—Ü—ñ—ó
            if file_path.stat().st_mtime > cutoff_time:
                # –ó–±–µ—Ä–µ–≥—Ç–∏ —Å—Ç—Ä—É–∫—Ç—É—Ä—É –¥–∏—Ä–µ–∫—Ç–æ—Ä—ñ–π
                relative_path = file_path.relative_to(source)
                dest_path = backup / relative_path

                # –°—Ç–≤–æ—Ä–∏—Ç–∏ –±–∞—Ç—å–∫—ñ–≤—Å—å–∫—ñ –¥–∏—Ä–µ–∫—Ç–æ—Ä—ñ—ó
                dest_path.parent.mkdir(parents=True, exist_ok=True)

                # –ö–æ–ø—ñ—é–≤–∞—Ç–∏ –∑ –º–µ—Ç–∞–¥–∞–Ω–∏–º–∏
                shutil.copy2(file_path, dest_path)
                copied_count += 1

    print(f"–°–∫–æ–ø—ñ–π–æ–≤–∞–Ω–æ {copied_count} —Ñ–∞–π–ª—ñ–≤")

# –í–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–Ω—è
backup_recent_files("/home/user/documents", "/backup/documents", days=7)

import shutil

def rotate_files(base_name, max_count=5):
    """
    –†–æ—Ç—É–≤–∞—Ç–∏ —Ñ–∞–π–ª–∏: file.log ‚Üí file.log.1 ‚Üí file.log.2 ...
    –ó–±–µ—Ä—ñ–≥–∞—Ç–∏ –º–∞–∫—Å–∏–º—É–º max_count —Å—Ç–∞—Ä–∏—Ö –≤–µ—Ä—Å—ñ–π
    """

    oldest = Path(f"{base_name}.{max_count}")

        new_file = Path(f"{base_name}.{i + 1}")

        if old_file.exists():
            shutil.move(old_file, new_file)

    if base_path.exists():
        shutil.move(base_path, f"{base_name}.1")

    # –°—Ç–≤–æ—Ä–∏—Ç–∏ –Ω–æ–≤–∏–π —Ñ–∞–π–ª
    base_path.touch()

    print(f"–†–æ—Ç–∞—Ü—ñ—è –∑–∞–≤–µ—Ä—à–µ–Ω–∞. –ó–±–µ—Ä—ñ–≥–∞—î—Ç—å—Å—è {max_count} –≤–µ—Ä—Å—ñ–π.")

import shutil

def rotate_and_compress(log_file, keep_days=7):
    """
    –†–æ—Ç—É–≤–∞—Ç–∏ —Ç–∞ —Å—Ç–∏—Å–Ω—É—Ç–∏ —Å—Ç–∞—Ä—ñ –ª–æ–≥–∏
    –í–∏–¥–∞–ª—è—Ç–∏ –ª–æ–≥–∏ —Å—Ç–∞—Ä—à—ñ –∑–∞ keep_days
    """

        return

    today = datetime.date.today()
    dated_name = f"{log_path.stem}.{today}.log"
    dated_path = log_path.parent / dated_name

    # –ü–µ—Ä–µ–º—ñ—Å—Ç–∏—Ç–∏ –ø–æ—Ç–æ—á–Ω–∏–π –ª–æ–≥
    if log_path.stat().st_size > 0:

        # –°—Ç–∏—Å–Ω—É—Ç–∏
        gz_path = Path(f"{dated_path}.gz")
        with open(dated_path, 'rb') as f_in:
            with gzip.open(gz_path, 'wb') as f_out:
                shutil.copyfileobj(f_in, f_out)

        # –í–∏–¥–∞–ª–∏—Ç–∏ –Ω–µ—Å—Ç–∏—Å–Ω—É—Ç–∏–π
        dated_path.unlink()
        print(f"–õ–æ–≥ —Å—Ç–∏—Å–Ω—É—Ç–æ: {gz_path.name}")

    # –°—Ç–≤–æ—Ä–∏—Ç–∏ –Ω–æ–≤–∏–π –ª–æ–≥
    log_path.touch()

    # –û—á–∏—Å—Ç–∏—Ç–∏ —Å—Ç–∞—Ä—ñ –ª–æ–≥–∏
    cutoff_date = today - datetime.timedelta(days=keep_days)

        # –í–∏—Ç—è–≥—Ç–∏ –¥–∞—Ç—É –∑ —ñ–º–µ–Ω—ñ —Ñ–∞–π–ª—É
        try:
            date_str = old_log.stem.split('.')[-2]  # 2024-11-09
            log_date = datetime.datetime.strptime(date_str, "%Y-%m-%d").date()

            if log_date < cutoff_date:
                old_log.unlink()
                print(f"–í–∏–¥–∞–ª–µ–Ω–æ —Å—Ç–∞—Ä–∏–π –ª–æ–≥: {old_log.name}")
        except:
            pass

def compress_file(input_file, output_file=None):
    """–°—Ç–∏—Å–Ω—É—Ç–∏ —Ñ–∞–π–ª –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é gzip"""

### **–°–ª–∞–π–¥ 27. –ê—Ä—Ö—ñ—Ç–µ–∫—Ç—É—Ä–∞ backup —Å–∏—Å—Ç–µ–º–∏**

```python
from pathlib import Path
import datetime
import shutil
import tarfile
import logging

class BackupSystem:
    """–ê–≤—Ç–æ–º–∞—Ç–∏—á–Ω–∞ —Å–∏—Å—Ç–µ–º–∞ —Ä–µ–∑–µ—Ä–≤–Ω–æ–≥–æ –∫–æ–ø—ñ—é–≤–∞–Ω–Ω—è"""

    def __init__(self, source_dir, backup_dir, keep_count=7):
        self.source = Path(source_dir)
        self.backup_dir = Path(backup_dir)
        self.keep_count = keep_count

        # –°—Ç–≤–æ—Ä–∏—Ç–∏ –¥–∏—Ä–µ–∫—Ç–æ—Ä—ñ—é –¥–ª—è backup
        self.backup_dir.mkdir(parents=True, exist_ok=True)

        # –ù–∞–ª–∞—à—Ç—É–≤–∞—Ç–∏ –ª–æ–≥—É–≤–∞–Ω–Ω—è
        self._setup_logging()

    def _setup_logging(self):
        """–ù–∞–ª–∞—à—Ç—É–≤–∞—Ç–∏ –ª–æ–≥–µ—Ä"""
        log_file = self.backup_dir / "backup.log"
        logging.basicConfig(
            level=logging.INFO,
            format='%(asctime)s - %(levelname)s - %(message)s',
            handlers=[
                logging.FileHandler(log_file),
                logging.StreamHandler()
            ]
        )
        self.logger = logging.getLogger(__name__)

    def create_backup(self):
        """–°—Ç–≤–æ—Ä–∏—Ç–∏ —Ä–µ–∑–µ—Ä–≤–Ω—É –∫–æ–ø—ñ—é"""
        # –ì–µ–Ω–µ—Ä—É–≤–∞—Ç–∏ —ñ–º'—è –∑ timestamp
        timestamp = datetime.datetime.now().strftime("%Y-%m-%d_%H-%M-%S")
        backup_name = f"backup_{self.source.name}_{timestamp}.tar.gz"
        backup_path = self.backup_dir / backup_name

        self.logger.info(f"–ü–æ—á–∞—Ç–æ–∫ backup: {self.source} ‚Üí {backup_path}")

        # –ü–µ—Ä–µ–≤—ñ—Ä–∏—Ç–∏ –º—ñ—Å—Ü–µ –Ω–∞ –¥–∏—Å–∫—É
        if not self._check_disk_space():
            self.logger.error("–ù–µ–¥–æ—Å—Ç–∞—Ç–Ω—å–æ –º—ñ—Å—Ü—è –Ω–∞ –¥–∏—Å–∫—É!")
            return False

        try:
            # –°—Ç–≤–æ—Ä–∏—Ç–∏ –∞—Ä—Ö—ñ–≤
            with tarfile.open(backup_path, "w:gz") as tar:
                tar.add(self.source, arcname=self.source.name)

            size_mb = backup_path.stat().st_size / (1024 ** 2)
            self.logger.info(f"Backup —Å—Ç–≤–æ—Ä–µ–Ω–æ: {backup_name} ({size_mb:.2f} MB)")

            # –†–æ—Ç—É–≤–∞—Ç–∏ —Å—Ç–∞—Ä—ñ backup'–∏
            self._rotate_backups()

            return True

        except Exception as e:
            self.logger.error(f"–ü–æ–º–∏–ª–∫–∞ —Å—Ç–≤–æ—Ä–µ–Ω–Ω—è backup: {e}")
            return False

    def _check_disk_space(self, required_mb=100):
        """–ü–µ—Ä–µ–≤—ñ—Ä–∏—Ç–∏ –≤—ñ–ª—å–Ω–µ –º—ñ—Å—Ü–µ"""
        usage = shutil.disk_usage(self.backup_dir)
        free_mb = usage.free / (1024 ** 2)
        return free_mb > required_mb

    def _rotate_backups(self):
        """–í–∏–¥–∞–ª–∏—Ç–∏ —Å—Ç–∞—Ä—ñ backup'–∏"""
        # –ó–Ω–∞–π—Ç–∏ –≤—Å—ñ backup —Ñ–∞–π–ª–∏
        backups = sorted(
            self.backup_dir.glob("backup_*.tar.gz"),
            key=lambda x: x.stat().st_mtime,
            reverse=True
        )

        # –í–∏–¥–∞–ª–∏—Ç–∏ —Å—Ç–∞—Ä—ñ—à—ñ –∑–∞ keep_count
        for old_backup in backups[self.keep_count:]:
            old_backup.unlink()
            self.logger.info(f"–í–∏–¥–∞–ª–µ–Ω–æ —Å—Ç–∞—Ä–∏–π backup: {old_backup.name}")

    def restore_backup(self, backup_file, restore_to):
        """–í—ñ–¥–Ω–æ–≤–∏—Ç–∏ –∑ backup"""
        backup_path = self.backup_dir / backup_file
        restore_path = Path(restore_to)

        self.logger.info(f"–í—ñ–¥–Ω–æ–≤–ª–µ–Ω–Ω—è: {backup_file} ‚Üí {restore_to}")

        try:
            with tarfile.open(backup_path, "r:gz") as tar:
                tar.extractall(restore_path)

            self.logger.info("–í—ñ–¥–Ω–æ–≤–ª–µ–Ω–Ω—è —É—Å–ø—ñ—à–Ω–µ")
            return True

        except Exception as e:
            self.logger.error(f"–ü–æ–º–∏–ª–∫–∞ –≤—ñ–¥–Ω–æ–≤–ª–µ–Ω–Ω—è: {e}")
            return False

    def list_backups(self):
        """–ü–æ–∫–∞–∑–∞—Ç–∏ —Å–ø–∏—Å–æ–∫ backup'—ñ–≤"""
        backups = sorted(
            self.backup_dir.glob("backup_*.tar.gz"),
            key=lambda x: x.stat().st_mtime,
            reverse=True
        )

        print(f"\n–î–æ—Å—Ç—É–ø–Ω—ñ backup'–∏ ({len(backups)}):")
        print("-" * 70)
        print(f"{'–§–∞–π–ª':<40} {'–î–∞—Ç–∞':>20} {'–†–æ–∑–º—ñ—Ä':>10}")
        print("-" * 70)

        for backup in backups:
            mtime = datetime.datetime.fromtimestamp(backup.stat().st_mtime)
            size_mb = backup.stat().st_size / (1024 ** 2)
            print(f"{backup.name:<40} {mtime.strftime('%Y-%m-%d %H:%M:%S'):>20} {size_mb:>8.2f} MB")
```

### **–°–ª–∞–π–¥ 29. –†–æ–∑—à–∏—Ä–µ–Ω–Ω—è: –Ü–Ω–∫—Ä–µ–º–µ–Ω—Ç–∞–ª—å–Ω–∏–π backup**

```python
import hashlib
import json

class IncrementalBackup(BackupSystem):
    """Backup –∑ –ø—ñ–¥—Ç—Ä–∏–º–∫–æ—é —ñ–Ω–∫—Ä–µ–º–µ–Ω—Ç–∞–ª—å–Ω–∏—Ö –∫–æ–ø—ñ–π"""

    def __init__(self, *args, **kwargs):
        super().__init__(*args, **kwargs)
        self.manifest_file = self.backup_dir / "manifest.json"
        self.manifest = self._load_manifest()

    def _load_manifest(self):
        """–ó–∞–≤–∞–Ω—Ç–∞–∂–∏—Ç–∏ –º–∞–Ω—ñ—Ñ–µ—Å—Ç (—Å–ø–∏—Å–æ–∫ —Ñ–∞–π–ª—ñ–≤ + hash)"""
        if self.manifest_file.exists():
            with open(self.manifest_file, 'r') as f:
                return json.load(f)
        return {}

    def _save_manifest(self):
        """–ó–±–µ—Ä–µ–≥—Ç–∏ –º–∞–Ω—ñ—Ñ–µ—Å—Ç"""
        with open(self.manifest_file, 'w') as f:
            json.dump(self.manifest, f, indent=2)

    def _calculate_hash(self, file_path):
        """–û–±—á–∏—Å–ª–∏—Ç–∏ MD5 hash —Ñ–∞–π–ª—É"""
        hasher = hashlib.md5()
        with open(file_path, 'rb') as f:
            for chunk in iter(lambda: f.read(4096), b''):
                hasher.update(chunk)
        return hasher.hexdigest()

    def create_incremental_backup(self):
        """–°—Ç–≤–æ—Ä–∏—Ç–∏ —ñ–Ω–∫—Ä–µ–º–µ–Ω—Ç–∞–ª—å–Ω–∏–π backup (—Ç—ñ–ª—å–∫–∏ –∑–º—ñ–Ω–µ–Ω—ñ —Ñ–∞–π–ª–∏)"""
        timestamp = datetime.datetime.now().strftime("%Y-%m-%d_%H-%M-%S")
        backup_name = f"incremental_{timestamp}.tar.gz"
        backup_path = self.backup_dir / backup_name

        changed_files = []

        # –ó–Ω–∞–π—Ç–∏ –∑–º—ñ–Ω–µ–Ω—ñ —Ñ–∞–π–ª–∏
        for file in self.source.rglob("*"):
            if file.is_file():
                file_hash = self._calculate_hash(file)
                relative_path = str(file.relative_to(self.source))

                # –ü–µ—Ä–µ–≤—ñ—Ä–∏—Ç–∏, —á–∏ —Ñ–∞–π–ª –∑–º—ñ–Ω–∏–≤—Å—è
                if relative_path not in self.manifest or self.manifest[relative_path] != file_hash:
                    changed_files.append(file)
                    self.manifest[relative_path] = file_hash

        if not changed_files:
            self.logger.info("–ù–µ–º–∞—î –∑–º—ñ–Ω–µ–Ω–∏—Ö —Ñ–∞–π–ª—ñ–≤")
            return

        # –°—Ç–≤–æ—Ä–∏—Ç–∏ –∞—Ä—Ö—ñ–≤ —Ç—ñ–ª—å–∫–∏ –∑—ñ –∑–º—ñ–Ω–µ–Ω–∏–º–∏ —Ñ–∞–π–ª–∞–º–∏
        with tarfile.open(backup_path, "w:gz") as tar:
            for file in changed_files:
                arcname = file.relative_to(self.source)
                tar.add(file, arcname=arcname)

        self._save_manifest()
        self.logger.info(f"–Ü–Ω–∫—Ä–µ–º–µ–Ω—Ç–∞–ª—å–Ω–∏–π backup: {len(changed_files)} —Ñ–∞–π–ª—ñ–≤")
```
