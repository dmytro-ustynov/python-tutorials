class IPAddress:
    """–ö–ª–∞—Å –¥–ª—è —Ä–æ–±–æ—Ç–∏ –∑ IPv4 –∞–¥—Ä–µ—Å–∞–º–∏"""
    
    def __init__(self, ip_string):
        """–Ü–Ω—ñ—Ü—ñ–∞–ª—ñ–∑–∞—Ü—ñ—è –æ–±'—î–∫—Ç–∞ IP-–∞–¥—Ä–µ—Å–∏"""
        parts = ip_string.split('.')
        if len(parts) != 4:
            raise ValueError("Invalid IP address format")
        
        self.octets = [int(octet) for octet in parts]
        
        # –í–∞–ª—ñ–¥–∞—Ü—ñ—è –¥—ñ–∞–ø–∞–∑–æ–Ω—É
        for octet in self.octets:
            if not 0 <= octet <= 255:
                raise ValueError(f"Octet {octet} out of range (0-255)")
    
    def __str__(self):
        """–î–ª—è print() —Ç–∞ str() ‚Äî –∑—Ä—É—á–Ω–∏–π –≤–∏–≥–ª—è–¥ –¥–ª—è –∫–æ—Ä–∏—Å—Ç—É–≤–∞—á–∞"""
        return '.'.join(map(str, self.octets))
    
    def __repr__(self):
        """–î–ª—è —Ä–æ–∑—Ä–æ–±–Ω–∏–∫—ñ–≤ ‚Äî –ø–æ–∫–∞–∑—É—î, —è–∫ —Å—Ç–≤–æ—Ä–∏—Ç–∏ –æ–±'—î–∫—Ç"""
        return f"IPAddress('{self}')"
    
    def __eq__(self, other):
        """–û–ø–µ—Ä–∞—Ç–æ—Ä == –¥–ª—è –ø–æ—Ä—ñ–≤–Ω—è–Ω–Ω—è IP-–∞–¥—Ä–µ—Å"""
        if not isinstance(other, IPAddress):
            return False
        return self.octets == other.octets
    
    def __lt__(self, other):
        """–û–ø–µ—Ä–∞—Ç–æ—Ä < –¥–ª—è –ø–æ—Ä—ñ–≤–Ω—è–Ω–Ω—è IP-–∞–¥—Ä–µ—Å"""
        if not isinstance(other, IPAddress):
            return NotImplemented
        return self.octets < other.octets
    
    def __hash__(self):
        """–î–æ–∑–≤–æ–ª—è—î –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É–≤–∞—Ç–∏ IP –≤ –º–Ω–æ–∂–∏–Ω–∞—Ö —Ç–∞ —è–∫ –∫–ª—é—á –≤ —Å–ª–æ–≤–Ω–∏–∫–∞—Ö"""
        return hash(tuple(self.octets))
    
    def __int__(self):
        """–ö–æ–Ω–≤–µ—Ä—Ç–∞—Ü—ñ—è IP –≤ —á–∏—Å–ª–æ (–¥–ª—è –ø—ñ–¥—Å—É–º–æ–≤—É–≤–∞–Ω–Ω—è)"""
        return (self.octets[0] << 24) + (self.octets[1] << 16) + \
               (self.octets[2] << 8) + self.octets[3]
    
    def __add__(self, value):
        """–î–æ–¥–∞–≤–∞–Ω–Ω—è —á–∏—Å–ª–∞ –¥–æ IP-–∞–¥—Ä–µ—Å–∏ (—ñ–Ω–∫—Ä–µ–º–µ–Ω—Ç)"""
        if not isinstance(value, int):
            return NotImplemented
        
        new_value = int(self) + value
        if new_value > 0xFFFFFFFF:  # –ú–∞–∫—Å–∏–º–∞–ª—å–Ω–∞ IPv4 –∞–¥—Ä–µ—Å–∞
            raise ValueError("IP address overflow")

                        f"{(new_value >> 16) & 0xFF}."

class Encryptor(ABC):
    """–ê–±—Å—Ç—Ä–∞–∫—Ç–Ω–∏–π –±–∞–∑–æ–≤–∏–π –∫–ª–∞—Å –¥–ª—è —à–∏—Ñ—Ä—É–≤–∞–Ω–Ω—è"""

class CaesarCipher(Encryptor):
    """–®–∏—Ñ—Ä –¶–µ–∑–∞—Ä—è - –ø—Ä–æ—Å—Ç–∏–π –ø—ñ–¥—Å—Ç–∞–Ω–æ–≤–æ—á–Ω–∏–π —à–∏—Ñ—Ä (–ù–ï –±–µ–∑–ø–µ—á–Ω–∏–π!)"""
    
    def __init__(self, shift: int = 3):
        self.shift = shift
    
    def encrypt(self, plaintext: str) -> str:
        """–ó—Å—É–≤ –∫–æ–∂–Ω–æ—ó –ª—ñ—Ç–µ—Ä–∏ –Ω–∞ shift –ø–æ–∑–∏—Ü—ñ–π"""
        result = []
        for char in plaintext:
            if char.isalpha():
                # –í–∏–∑–Ω–∞—á–∞—î–º–æ –±–∞–∑—É (A –∞–±–æ a)
                base = ord('A') if char.isupper() else ord('a')
                # –ó—Å—É–≤–∞—î–º–æ –ª—ñ—Ç–µ—Ä—É
                shifted = (ord(char) - base + self.shift) % 26
                result.append(chr(base + shifted))
            else:
                result.append(char)
        return ''.join(result)
    
    def decrypt(self, ciphertext: str) -> str:
        """–ó–≤–æ—Ä–æ—Ç–Ω–∏–π –∑—Å—É–≤ –¥–ª—è —Ä–æ–∑—à–∏—Ñ—Ä—É–≤–∞–Ω–Ω—è"""
        # –ü—Ä–æ—Å—Ç–æ —à–∏—Ñ—Ä—É—î–º–æ –∑ –Ω–µ–≥–∞—Ç–∏–≤–Ω–∏–º –∑—Å—É–≤–æ–º
        original_shift = self.shift
        self.shift = -self.shift
        result = self.encrypt(ciphertext)
        self.shift = original_shift
        return result
    
    def __str__(self):
        return f"CaesarCipher(shift={self.shift})"


class XORCipher(Encryptor):
    """XOR —à–∏—Ñ—Ä—É–≤–∞–Ω–Ω—è - –∫–æ–∂–µ–Ω –±–∞–π—Ç XOR –∑ –∫–ª—é—á–µ–º"""
    
    def __init__(self, key: str):
        self.key = key.encode()
    
    def _xor_bytes(self, data: bytes) -> bytes:
        """XOR –æ–ø–µ—Ä–∞—Ü—ñ—è –º—ñ–∂ –¥–∞–Ω–∏–º–∏ —Ç–∞ –∫–ª—é—á–µ–º"""
        return bytes([
            data[i] ^ self.key[i % len(self.key)]
            for i in range(len(data))
        ])
    
    def encrypt(self, plaintext: str) -> str:
        """–®–∏—Ñ—Ä—É–≤–∞–Ω–Ω—è —á–µ—Ä–µ–∑ XOR"""
        data = plaintext.encode('utf-8')
        encrypted = self._xor_bytes(data)
        # –ö–æ–¥—É—î–º–æ –≤ base64 –¥–ª—è –∑—Ä—É—á–Ω–æ—Å—Ç—ñ –ø–µ—Ä–µ–¥–∞—á—ñ
        return base64.b64encode(encrypted).decode('utf-8')
    
    def decrypt(self, ciphertext: str) -> str:
        """–†–æ–∑—à–∏—Ñ—Ä—É–≤–∞–Ω–Ω—è —á–µ—Ä–µ–∑ XOR (—Ç–∞ —Å–∞–º–∞ –æ–ø–µ—Ä–∞—Ü—ñ—è!)"""
        encrypted = base64.b64decode(ciphertext.encode('utf-8'))
        decrypted = self._xor_bytes(encrypted)
        return decrypted.decode('utf-8')
    
    def __str__(self):
        return f"XORCipher(key_length={len(self.key)})"


class FernetEncryptor(Encryptor):
    """–ë–µ–∑–ø–µ—á–Ω–µ —Å–∏–º–µ—Ç—Ä–∏—á–Ω–µ —à–∏—Ñ—Ä—É–≤–∞–Ω–Ω—è –∑ –±—ñ–±–ª—ñ–æ—Ç–µ–∫–∏ cryptography"""

        """–ì–µ–Ω–µ—Ä–∞—Ü—ñ—è –∫–ª—é—á–∞ –∑ –ø–∞—Ä–æ–ª—è —á–µ—Ä–µ–∑ KDF"""
        # –í–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—î–º–æ PBKDF2 –¥–ª—è —Å—Ç–≤–æ—Ä–µ–Ω–Ω—è —Å–∏–ª—å–Ω–æ–≥–æ –∫–ª—é—á–∞ –∑ –ø–∞—Ä–æ–ª—è
        kdf = PBKDF2(
            algorithm=hashes.SHA256(),
            length=32,
            salt=b'static_salt_change_me',  # –í –ø—Ä–æ–¥–∞–∫—à–µ–Ω—ñ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É–≤–∞—Ç–∏ –≤–∏–ø–∞–¥–∫–æ–≤–∏–π salt!
            iterations=100000,
        )
        key = base64.urlsafe_b64encode(kdf.derive(password.encode()))
        self.cipher = Fernet(key)
    
    def encrypt(self, plaintext: str) -> str:
        """–ë–µ–∑–ø–µ—á–Ω–µ —à–∏—Ñ—Ä—É–≤–∞–Ω–Ω—è"""
        return self.cipher.encrypt(plaintext.encode()).decode()

        """–ë–µ–∑–ø–µ—á–Ω–µ —Ä–æ–∑—à–∏—Ñ—Ä—É–≤–∞–Ω–Ω—è"""
        return self.cipher.decrypt(ciphertext.encode()).decode()
    
    def __str__(self):
        return "FernetEncryptor(AES-128 in CBC mode)"
